#include <stdio.h>
#include <stdlib.h>

void error(char *str);

int main(int argc,char *argv[]){

	char ch;
	FILE *fd = NULL;
	
	if(argc >= 2){
		if(! (fd = fopen(argv[1],"r")) )
			error("open file");
	}
	else{
		error("is few arguments");
	}

	while( (ch = getc(fd)) != EOF )
		putc(ch,stdout);

	if(fd != NULL)
		fclose(fd);

	return 0;
}

void error(char *str){
	printf("[error]: %s\n",str);
	exit(1);
}
